include(ExternalProject)

	nlopt
	#GIT_REPOSITORY git://github.com/stevengj/nlopt.git
	#GIT_TAG 2.5.0
	URL https://github.com/stevengj/nlopt/archive/v2.5.0.zip

	URL_MD5 229ac19157563c76b1b1c28ef3f3f7d0

	INSTALL_DIR ${CMAKE_INSTALL_PREFIX}
